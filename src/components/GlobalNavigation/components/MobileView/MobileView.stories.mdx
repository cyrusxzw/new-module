      const [loginOpen, setLoginOpen] = React.useState(false);
      const handleOnLoginClick = () => setLoginOpen(true);

        'Presentation',
      );
      const theme = knobs.select(
        'theme',
        ['dark', 'light'],
        'dark',
        'Presentation',
      );
      const implementation = {
        gridOverlay: {
          isActive: knobs.boolean('Show grid overlay', false, 'Implementation'),
          hasInvertedColours: knobs.boolean(
            'Invert grid overlay colours',
            false,
            'Implementation',
          ),
        },
      };
      const fixture = { ...GlobalNavigationFixture };
      fixture.actions.cart.onClick = handleOnCartClick;
      return (
        <>
          <GlobalNavigationStateContextProvider>
            <GlobalNavigationContextProvider
              value={{
                actions: fixture.actions,
                read: fixture.read,
                collections: fixture.collections,
                isVisuallyObstructed: cartOpen || loginOpen || cartOpen,
                mobileViewClosedTheme,
                theme,
              }}
            >
              <MobileViewContextProvider>
                <MobileView />
              </MobileViewContextProvider>
            </GlobalNavigationContextProvider>
          </GlobalNavigationStateContextProvider>
          <Modal
            copy={{
              close: 'Close Modal',

            onClose={handleOnCartCloseClick}
          >

          </Modal>
          <StorybookGridOverlay {...implementation.gridOverlay} />

      );
    })()}
