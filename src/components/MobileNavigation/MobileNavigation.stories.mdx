# MobileNavigation

<Canvas>
  <Story name="Base component">
    {(() => {
      const [cartOpen, setCartOpen] = React.useState(false);
      const handleOnCartClick = () => setCartOpen(true);
      const handleOnCartCloseClick = () => setCartOpen(false);
      const [searchOpen, setSearchOpen] = React.useState(false);
      const handleOnSearchClick = () => setSearchOpen(true);
      const handleOnSearchCloseClick = () => setSearchOpen(false);
      const [loginOpen, setLoginOpen] = React.useState(false);
      const handleOnLoginClick = () => setLoginOpen(true);
      const handleOnLoginCloseClick = () => setLoginOpen(false);
      const closedTheme = knobs.select(
        'closedTheme',
        ['dark', 'light'],
        'dark',
        'Presentation',
      );
      const theme = knobs.select(
        'theme',
        ['dark', 'light'],
        'dark',
        'Presentation',
      );
      const implementation = {
        gridOverlay: {
          isActive: knobs.boolean('Show grid overlay', false, 'Implementation'),
          hasInvertedColours: knobs.boolean(
            'Invert grid overlay colours',
            false,
            'Implementation',
          ),
        },
      };
      const fixture = { ...MobileNavigationFixture };
      fixture.header.cart.onClick = handleOnCartClick;
      fixture.header.search.onClick = handleOnSearchClick;
      return (
        <>
          <MobileNavigation
            articles={fixture.articles}
            closedTheme={closedTheme}
            header={fixture.header}
            isVisuallyObstructed={cartOpen || loginOpen || cartOpen}
            items={fixture.items}
            secondaryItems={fixture.secondaryItems}
            theme={theme}
          />
          <Modal
            copy={{
              close: 'Close Modal',
            }}
            isVisible={cartOpen}
            onClose={handleOnCartCloseClick}
          >
            You have 0 items in you cart.
          </Modal>
          <Modal
            copy={{
              close: 'Close Modal',
            }}
            isVisible={searchOpen}
            onClose={handleOnSearchCloseClick}
          >
            Search component
          </Modal>
          <StorybookGridOverlay {...implementation.gridOverlay} />
        </>
      );
    })()}
  </Story>
</Canvas>
