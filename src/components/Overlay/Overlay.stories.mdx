# Overlay

<Canvas>
  <Story name="Base component">
    <State initialState={{ isVisible: false }}>
      {(state, setState) => {
        return (
          <React.Fragment>
            <Overlay
              isVisible={state.isVisible}
              onClose={() => {
                setState(() => ({ ...state, isVisible: false }));
              }}
            />

              onClick={() => {
                setState(() => ({ ...state, isVisible: true }));
              }}
              title="Open Overlay"

          </React.Fragment>
        );
      }}
    </State>
