# Modal

<Canvas>
  <Story name="Base component">
    <State initialState={{ isVisible: false }}>
      {(state, setState) => {
        const handleOnClick = () => {
          setState(() => ({ ...state, isVisible: true }));
        };
        const handleOnClose = () => {
          setState(() => ({ ...state, isVisible: false }));
        };
        return (
          <React.Fragment>
            <Modal
              copy={ModalFixture.copy}
              isVisible={state.isVisible}
              onClose={handleOnClose}
            >
              {ModalFixture.children}
            </Modal>
            <Button onClick={handleOnClick}>Open Modal</Button>
          </React.Fragment>
        );
      }}
    </State>
  </Story>
